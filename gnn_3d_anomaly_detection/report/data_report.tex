\section{Node and Edge Features}

\subsection{Node Features}
Each node contains 6 features:
\begin{itemize}
    \item $x$: X-coordinate (normalized)
    \item $y$: Y-coordinate (normalized)
    \item $timestamp$: Temporal coordinate (normalized per-sample to [0,1])
    \item $width$: Bounding box width
    \item $height$: Bounding box height
    \item $vehicle\_id$: Vehicle identifier
\end{itemize}

\subsection{Edge Features}
Each edge contains 3 features:
\begin{itemize}
    \item $distance$: Euclidean distance between vehicle centroids
    \item $frame$: Frame number
    \item $timestamp$: Timestamp of the frame
\end{itemize}

\section{Model Architecture Details}

\subsection{Layer Dimensions}
\begin{itemize}
    \item Input: $N \times 6$ (N nodes, 6 features)
    \item GCN Layer 1: $6 \to 64$
    \item GCN Layer 2: $64 \to 64$
    \item GCN Layer 3: $64 \to 64$
    \item Pooling output: $192$ (64 $\times$ 3: mean + max + sum)
    \item FC Layer 1: $192 \to 64$
    \item FC Layer 2: $64 \to 32$
    \item FC Layer 3: $32 \to 1$
\end{itemize}

\subsection{Activation and Regularization}
\begin{itemize}
    \item Activation: ReLU (after layers 1-2, none after layer 3)
    \item Dropout: 0.5 (after GCN layers 1-2, after FC layers 1-2)
    \item Pooling: Global mean, max, and sum (concatenated)
    \item Output activation: Sigmoid (for probability)
\end{itemize}

\subsection{Key Findings}

\begin{itemize}
    \item \textbf{Simpler architecture performs better:} Best configuration uses only 2 GCN layers (vs. baseline 3)
    \item \textbf{Lower hidden dimension:} Best configuration uses 32 (vs. baseline 64)
    \item \textbf{Lower dropout:} Best configuration uses 0.1 (vs. baseline 0.5)
    \item \textbf{Sum pooling effective:} Best configuration uses sum pooling (vs. baseline all pooling types)
    \item \textbf{Higher batch size:} Best configuration uses batch size 6 (vs. baseline 1)
    \item \textbf{Improvement:} Validation accuracy improved from 66.67\% (baseline) to 83.33\% (best), a 16.66 percentage point improvement
\end{itemize}

\subsection{Optimization Methodology}

\begin{itemize}
    \item \textbf{Framework:} Optuna with TPE (Tree-structured Parzen Estimator) sampler
    \item \textbf{Storage:} SQLite database for persistence and resume capability
    \item \textbf{Objective:} Maximize validation accuracy
    \item \textbf{Early stopping:} Applied per trial with trial-specific patience and min\_delta
    \item \textbf{Resume support:} Optimization can be interrupted and resumed
\end{itemize}
